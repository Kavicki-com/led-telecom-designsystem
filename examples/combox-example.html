Como Usar

Exemplo de implementação em React:

import React from 'react'; import { Combox } from 'led-telecom'; function MyComponent() { const handleChange = (value: string) => { console.log('Valor alterado:', value); }; const handleOptionSelect = (option: string) => { console.log('Opção selecionada:', option); }; return ( <Combox label="Selecione uma opção" description="Escolha uma das opções disponíveis" options={['Opção 1', 'Opção 2', 'Opção 3']} onChange={handleChange} onOptionSelect={handleOptionSelect} /> ); }

Props Disponíveis

Lista de todas as propriedades do componente:

interface ComboxProps { label?: string; // Label do combox inputText?: string; // Texto do input description?: string; // Descrição do combox showDescription?: boolean; // Se deve exibir a descrição showLabel?: boolean; // Se deve exibir o label state?: 'default' | 'open' | 'disable'; // Estado do combox options?: string[]; // Opções do dropdown selectedOption?: string; // Opção selecionada placeholder?: string; // Placeholder do input disabled?: boolean; // Se o combox está desabilitado value?: string; // Valor do input onChange?: (value: string) => void; // Callback para mudança de valor onOptionSelect?: (option: string) => void; // Callback para seleção onFocus?: () => void; // Callback para foco onBlur?: () => void; // Callback para perda de foco className?: string; // Classes CSS adicionais name?: string; // Nome do combox id?: string; // ID do combox }
